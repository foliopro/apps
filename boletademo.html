// ============================================================ // 4) EXTRACTOR DE ALTA PRECISIÓN (TASK FORCE GOOGLE/META) // ============================================================ function extraerDatosCotizante(allPages) { const items = allPages[0].items; // Foco en la carátula const fullText = items.map(i => i.text).join(" "); // LÓGICA DE EQUIPO: Buscamos por patrón, no por etiqueta // 1. NOMBRE: El bloque más largo de mayúsculas en la parte superior let nombre = items .filter(i => i.y > 600) // Solo buscamos en la parte de arriba del PDF .find(i => i.text.length > 12 && i.text === i.text.toUpperCase() && !i.text.includes("NIT")) ?.text.trim() || "NOMBRE NO ENCONTRADO"; // 2. CÉDULA: El primer número de 7 a 10 dígitos que aparezca let cedula = fullText.match(/\b(\d{7,10})\b/)?.[1] || "0"; // 3. FECHA: El primer formato dd/mm/aaaa let fecha = fullText.match(/(\d{2}\/\d{2}\/\d{4})/)?.[1] || "00/00/0000"; // ESTRUCTURA PURA PARA EXCEL VBA const contenidoTxt = `${nombre}\n${cedula}\n${fecha}`; // RENDERIZADO DE INTERFAZ PROFESIONAL const wrap = document.createElement('div'); wrap.className = 'row'; wrap.style.border = "2px solid var(--accent)"; wrap.style.background = "rgba(34, 211, 238, 0.05)"; wrap.innerHTML = `

📦 Payload para Excel (VBA Ready)

${contenidoTxt}
`; const tablesContainer = document.getElementById('tables'); tablesContainer.insertBefore(wrap, tablesContainer.firstChild); document.getElementById('btnVBA').onclick = () => { const blob = new Blob([contenidoTxt], { type: 'text/plain;charset=utf-8' }); const a = document.createElement('a'); a.href = URL.createObjectURL(blob); a.download = 'cotizante.txt'; a.click(); }; }
